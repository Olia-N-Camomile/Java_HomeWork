таблица телефонов по юзерам:
use phone_numbers;
SELECT users.*, phones.phone_number, phones.id as ph_id 
FROM users left join phones
on phones.user_id=users.id;

таблица телефонов по комнатам:
select rooms.id, rooms.room_number, phones.phone_number, 
phones.id as ph_id from rooms left join phones
on rooms.phone_id=phones.id;

номер комнаты для пользователя с заданной фамилией:
use phone_numbers;
select rooms.room_number as номер_комнаты, room_for_user.username as фамилия from rooms inner join 
(select phones.id as ph_id, users.username as username from phones inner join users 
on phones.user_id=users.id and users.username='Dolin') as room_for_user
on rooms.phone_id=room_for_user.ph_id;
